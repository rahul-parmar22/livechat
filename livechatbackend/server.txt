// backend/controllers/messageController.js
const Message = require('../models/Message');

exports.deleteMessage = async (req, res) => {
  try {
    const message = await Message.findById(req.params.id);
    if (!message) return res.status(404).send('Message not found');

    if (message.sender !== req.user.username) {
      return res.status(403).send('You can only delete your own messages');
    }

    await Message.findByIdAndDelete(req.params.id);
    res.send('Message deleted successfully');
  } catch (err) {
    console.error(err);
    res.status(500).send('Server error');
  }
};

// backend/routes/messages.js (ya alag file)
const multer = require('multer');
const storage = multer.diskStorage({
  destination: function (req, file, cb) {
    cb(null, 'uploads/');  // server pe upload folder hona chahiye
  },
  filename: function (req, file, cb) {
    cb(null, Date.now() + '-' + file.originalname);
  }
});
const upload = multer({ storage });

router.post('/upload', auth, upload.single('image'), async (req, res) => {
  try {
    const newMsg = new Message({
      sender: req.user.username,
      content: '',  // empty content, image-only message
      imageUrl: req.file.path,  // path jaha image saved hai
      timestamp: new Date()
    });
    await newMsg.save();
    res.json(newMsg);
  } catch (err) {
    console.error(err);
    res.status(500).send('Upload failed');
  }
});

const [image, setImage] = useState(null);

const sendImage = () => {
  if (!image) return;
  const formData = new FormData();
  formData.append('image', image);

  axios.post('/messages/upload', formData, {
    headers: {
      'Content-Type': 'multipart/form-data',
      Authorization: `Bearer ${localStorage.getItem('token')}`
    }
  })
  .then(res => {
    setMessages(prev => [...prev, res.data]);
    setImage(null);
  })
  .catch(() => alert('Image upload failed'));
};

return (
  <div>
    {/* Existing message input and send button */}
    <input type="file" onChange={e => setImage(e.target.files[0])} />
    <button onClick={sendImage}>Send Image</button>

    {/* Show messages */}
    {messages.map(msg => (
      <div key={msg._id}>
        <strong>{msg.sender}</strong>: {msg.content}
        {msg.imageUrl && (
          <img src={`http://localhost:5000/${msg.imageUrl}`} alt="sent" style={{ maxWidth: 200 }} />
        )}
      </div>
    ))}
  </div>
);
